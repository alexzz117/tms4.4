            ROLE_NAME: '',

            INFO: ''

        }
        this.roleDialogVisible = true
      },

        this.multipleSelection = val;

        if (this.flag === 'add') {

        var self = this
        ajax.post('/cmc/role/add', this.roleDialogform, function (data) {
          self.$message('创建成功。')
          self.roleDialogVisible = false
          self.selRole()
        })
      },
      delRole() {
        var self = this
        var data = this.multipleSelection[0]
        ajax.post('/cmc/role/del', {
          roleId: data.ROLE_ID
        }, function (data) {
          self.$message('删除成功。')
          self.selRole()
        })
      },
      selRole() {
        var self = this;
        ajax.post('/cmc/role/list', {
          pageindex:1,
          pagesize:10
        }, function (data) {
          if (data.page) {
            self.bindGridData(data)
          }
        })
      },
      updateRole() {
        var self = this;
        ajax.post('/cmc/role/mod', this.roleDialogform, function (data) {
          self.$message('更新成功。')
          self.roleDialogVisible = false
          self.selRole()
        })
